    api_key=os.getenv("APIDECK_API_KEY", ""),

) as apideck:
    res = apideck.crm.opportunities.list(request={
        "service_id": "salesforce",

            "status": "Completed",

            "direction": apideck_unify.SortDirection.DESC,
        },
